let leads=[]; fetch("/leads") .then(res=>res.json()) .then(data=>{ leads=data.reverse(); document.getElementById("totalLeads").innerHTML=leads.length; document.getElementById("todayLeads").innerHTML=leads.length; document.getElementById("newLeads").innerHTML= leads.filter(x=>x.status==="New").length; renderTable(leads); }); function renderTable(list){ let html=""; list.forEach(lead=>{ html+=` ${lead.name} ${lead.mobile} ${lead.city} ${lead.market} ${lead.status} `; }); document.getElementById("leadTable").innerHTML=html; } document.getElementById("searchBox").addEventListener("keyup",function(){ const value=this.value.toLowerCase(); const filtered=leads.filter(x=> x.name.toLowerCase().includes(value) || x.mobile.includes(value) ); renderTable(filtered); });
